3. Menú d'usuari

La navegació amb fletxes permet moure's entre les opcions del menú. Està implementat això pel fet de que usem el role="menu" i els botons dins d'aquest menú són role="menuitem". 
// Exemple 3: Tancar menú amb Escape i navegar amb fletxes
function toggleMenu() {
  const menuContent = document.getElementById('menu-content');
  menuContent.classList.toggle('open');
  if (menuContent.classList.contains('open')) {
    const items = menuContent.querySelectorAll('[role="menuitem"]');
    items[0].focus();
  }
}

document.getElementById('menu-btn').addEventListener('keydown', function(event) {
  const menuContent = document.getElementById('menu-content');
  if ((event.key === 'ArrowDown' || event.key === 'Enter' || event.key === ' ') && !menuContent.classList.contains('open')) {
    event.preventDefault();
    toggleMenu();
  }
});

document.getElementById('menu-content').addEventListener('keydown', function(event) {
  const items = Array.from(this.querySelectorAll('[role="menuitem"]'));
  const currentIndex = items.indexOf(document.activeElement);
  if (event.key === 'ArrowDown') {
    event.preventDefault();
    const nextIndex = (currentIndex + 1) % items.length;
    items[nextIndex].focus();
  } else if (event.key === 'ArrowUp') {
    event.preventDefault();
    const prevIndex = (currentIndex - 1 + items.length) % items.length;
    items[prevIndex].focus();
  } else if (event.key === 'Escape') {
    this.classList.remove('open');
    document.getElementById('menu-btn').focus();
  }
});
